# iOS Skeleton Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** The iOS half of the walking skeleton: log in via Keycloak (OIDC-PKCE), generate a libsignal identity key pair on first launch, and send/receive an opaque ciphertext blob over an authenticated WebSocket — with the test suite running against a **mocked** `Service` (no live backend), behind SwiftLint + coverage gates.

**Architecture:** SwiftUI app following `app/CLAUDE.md` layering (App → UI Feature → Page → Data → Foundation), kept minimal for the skeleton and grown later. The network/WebSocket boundary is a `MessageService` **protocol** — the seam the tests mock (the app deliberately mocks dependencies; it does **not** use Testcontainers). The Xcode project is generated by **XcodeGen** from `project.yml` (the `.xcodeproj` is git-ignored).

**Tech Stack:** Swift / SwiftUI, Xcode 16+, iOS 17+, XcodeGen, SwiftLint, AppAuth-iOS (OIDC-PKCE), libsignal-client (SwiftPM), XCTest + Xcode coverage (xccov).

**Workflow:** Branch `feature/phase-0-ios-skeleton` off `main`. Follow root `CLAUDE.md` workflow + quality gates. **This plan cannot be compiled/verified in a non-macOS/Xcode environment** — run every `xcodegen`, `xcodebuild`, and `swiftlint` step on a Mac with Xcode 16+ and report real output; do not mark steps complete on assumption.

**Scope note:** Single app module + test bundle. Real Signal sessions, the OneUI design system, and the full multi-module graph arrive in later slices. The app sends an **opaque ciphertext blob** (the skeleton does not yet do real E2EE); libsignal is included only to generate an identity key pair (dependency de-risking).

### Task 2: Coverage gate script (≥90%, exclusions)

**Files:** `app/scripts/coverage.sh`

- [ ] **Step 1: Write the coverage check**

Create `app/scripts/coverage.sh` (runs tests with coverage, fails below 90% on meaningful files; excludes pure-view/app-entry/generated):
```bash
#!/usr/bin/env bash
# Runs the Cloak test suite with coverage and fails if line coverage on
# meaningful code is below the threshold. Pure SwiftUI views, the app entry,
# and generated code are excluded (root CLAUDE.md > quality gates).
set -euo pipefail

THRESHOLD=${COVERAGE_THRESHOLD:-90}
EXCLUDE_REGEX='(View\.swift|CloakApp\.swift|\.generated\.swift)$'
DERIVED=build/derived

xcodebuild test \
  -scheme Cloak \
  -destination 'platform=iOS Simulator,name=iPhone 15' \
  -derivedDataPath "$DERIVED" \
  -enableCodeCoverage YES \
  -quiet

PROFDATA=$(find "$DERIVED" -name '*.xcresult' | head -1)
REPORT=$(xcrun xccov view --report --json "$PROFDATA")

# Aggregate covered/executable lines across files not matching the exclude regex.
python3 - "$REPORT" "$EXCLUDE_REGEX" "$THRESHOLD" <<'PY'
import json, re, sys
report = json.loads(sys.argv[1]); pattern = re.compile(sys.argv[2]); threshold = float(sys.argv[3])
covered = executable = 0
for target in report.get("targets", []):
    for f in target.get("files", []):
        if pattern.search(f["name"]):
            continue
        covered += f["coveredLines"]; executable += f["executableLines"]
pct = (covered / executable * 100) if executable else 100.0
print(f"Coverage (meaningful files): {pct:.1f}% ({covered}/{executable})")
if pct < threshold:
    print(f"FAIL: below {threshold}%"); sys.exit(1)
print("PASS")
PY
```
Make it executable: `chmod +x app/scripts/coverage.sh`.

> Note: `xccov view --report --json` takes the `.xcresult`; the snippet locates it under the derived-data path. Adjust the simulator name to one installed locally.

```swift
/// Wire envelope (matches docs/contracts/phase0-message-envelope.md).
/// `ciphertext` is base64 of opaque bytes; the client base64-decodes for display.
struct MessageEnvelope: Codable, Equatable, Sendable {
    let messageId: String
    let toSub: String
    let deviceId: String?

/// The network/WebSocket boundary. Tests mock this; the app never talks to a

    func connect(accessToken: String) async throws
    func send(_ envelope: MessageEnvelope) async throws
    /// Stream of envelopes delivered by the server.
    var incoming: AsyncStream<MessageEnvelope> { get }

}
```

```swift
import Foundation

final class WebSocketMessageService: NSObject, MessageService, @unchecked Sendable {
    private let url: URL
    private var task: URLSessionWebSocketTask?
    private var continuation: AsyncStream<MessageEnvelope>.Continuation?
    let incoming: AsyncStream<MessageEnvelope>

    init(url: URL) {
        self.url = url
        var cont: AsyncStream<MessageEnvelope>.Continuation!
        self.incoming = AsyncStream { cont = $0 }
        self.continuation = cont
        super.init()
    }

    static func encode(_ env: MessageEnvelope) throws -> String {
        String(decoding: try JSONEncoder().encode(env), as: UTF8.self)
    }

    func connect(accessToken: String) async throws {
        var request = URLRequest(url: url)
        request.setValue("Bearer \(accessToken)", forHTTPHeaderField: "Authorization")
        let session = URLSession(configuration: .default)
        let task = session.webSocketTask(with: request)
        self.task = task
        task.resume()
        receiveLoop()
    }

    func send(_ envelope: MessageEnvelope) async throws {
        try await task?.send(.string(Self.encode(envelope)))
    }

    private func receiveLoop() {
        task?.receive { [weak self] result in
            guard let self else { return }
            if case let .success(.string(text)) = result,
               let data = text.data(using: .utf8),
               let env = try? JSONDecoder().decode(MessageEnvelope.self, from: data) {
                self.continuation?.yield(env)
            }
            if case .success = result { self.receiveLoop() }
        }
    }

    func disconnect() {
        task?.cancel(with: .goingAway, reason: nil)
        continuation?.finish()
    }
}
```

```swift
import Foundation
import LibSignalClient

/// Abstracts where the private identity bytes live (Keychain in production).
protocol KeyStorage: AnyObject {
    func data(forKey key: String) -> Data?
    func set(_ data: Data, forKey key: String)
}

final class InMemoryKeyStorage: KeyStorage {
    private var store: [String: Data] = [:]
    func data(forKey key: String) -> Data? { store[key] }
    func set(_ data: Data, forKey key: String) { store[key] = data }
}

struct DeviceIdentity {
    let keyPair: IdentityKeyPair
    var publicKeyData: Data { Data(keyPair.identityKey.publicKey.serialize()) }
}

final class DeviceKeyStore {
    private let storage: KeyStorage
    private let key = "cloak.identityKeyPair"

    init(storage: KeyStorage) { self.storage = storage }

    /// Loads the persisted identity, generating + storing one on first launch.
    func loadOrCreateIdentity() throws -> DeviceIdentity {
        if let raw = storage.data(forKey: key) {
            return DeviceIdentity(keyPair: try IdentityKeyPair(bytes: Array(raw)))
        }
        let generated = IdentityKeyPair.generate()
        storage.set(Data(generated.serialize()), forKey: key)
        return DeviceIdentity(keyPair: generated)
    }
}
```
> This is the dependency-de-risking step: it links libsignal and exercises real key generation on-device. (Production swaps `InMemoryKeyStorage` for a Keychain-backed `KeyStorage`.)

```swift
import Foundation
import AppAuth
import UIKit

protocol AuthService: Sendable {
    /// Runs OIDC-PKCE against Keycloak; returns an access token.
    func login(presenting: UIViewController) async throws -> String
}

/// AppAuth-based OIDC-PKCE login against the Keycloak `cloak-ios` client.
final class KeycloakAuthService: AuthService {
    private let issuer: URL
    private let clientId = "cloak-ios"
    private let redirect = URL(string: "com.cloak.app://oauth-callback")!
    private var session: OIDExternalUserAgentSession?

    init(issuer: URL) { self.issuer = issuer }

    func login(presenting: UIViewController) async throws -> String {
        let config = try await withCheckedThrowingContinuation { cont in
            OIDAuthorizationService.discoverConfiguration(forIssuer: issuer) { cfg, err in
                if let cfg { cont.resume(returning: cfg) } else { cont.resume(throwing: err!) }
            }
        }
        let request = OIDAuthorizationRequest(
            configuration: config, clientId: clientId, clientSecret: nil,
            scopes: [OIDScopeOpenID, OIDScopeProfile], redirectURL: redirect,
            responseType: OIDResponseTypeCode, additionalParameters: nil)
        return try await withCheckedThrowingContinuation { cont in
            session = OIDAuthState.authState(byPresenting: request, presenting: presenting) { state, err in
                if let token = state?.lastTokenResponse?.accessToken {
                    cont.resume(returning: token)
                } else {
                    cont.resume(throwing: err ?? NSError(domain: "auth", code: -1))
                }
            }
        }
    }
}
```

Create `app/Cloak/Feature/Chat/ChatViewModel.swift`:
```swift
import Foundation

struct ChatMessage: Identifiable, Equatable {
    let id = UUID()
    let text: String
    let mine: Bool
}

@MainActor
final class ChatViewModel: ObservableObject {
    @Published private(set) var timeline: [ChatMessage] = []
    private let service: MessageService
    private let accessToken: String
    private let mySub: String
    private var listenTask: Task<Void, Never>?

    init(service: MessageService, accessToken: String, mySub: String) {
        self.service = service
        self.accessToken = accessToken
        self.mySub = mySub
    }

    func start() async {
        try? await service.connect(accessToken: accessToken)
        listenTask = Task { [weak self] in
            guard let self else { return }
            for await env in service.incoming {
                if let data = Data(base64Encoded: env.ciphertext) {
                    await MainActor.run { self.timeline.append(.init(text: String(decoding: data, as: UTF8.self), mine: false)) }
                }
            }
        }
    }

    func send(text: String, to recipientSub: String) async {
        // Skeleton: the "ciphertext" is just the bytes, base64-encoded. Real Signal
        // encryption replaces this in a later slice.
        let blob = Data(text.utf8).base64EncodedString()
        let env = MessageEnvelope(messageId: UUID().uuidString, toSub: recipientSub, deviceId: nil, ciphertext: blob)
        try? await service.send(env)
        timeline.append(.init(text: text, mine: true))   // optimistic echo
    }
}
```
